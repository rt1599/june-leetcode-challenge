#include<bits/stdc++.h>
using namespace std;
void reverseString(vector<char>& s) {
    int i=0;
    int j=s.size()-1;
    while(i<j)   
    {if(i!=j)
    {swap(s[i],s[j]);
     i++;
     j--;
    }
    }
    return ;
    }
int main()
{int a[]=["H","a","n","n","a","h"]
reverseString(a);
return 0;
}
